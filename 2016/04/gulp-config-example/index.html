gulpfile.js 配置

// 引入 gulp & 组件
var gulp = require('gulp');
var jshint = require('gulp-jshint');
var concat = require('gulp-concat');
var rename = require('gulp-rename');
var uglify = require('gulp-uglify');

//var notify = require('gulp-notify');
var del = require('del');

//var sass = require('gulp-sass');
//var less = require('gulp-less');

var minifyHTML = require('gulp-minify-html');  //代码压缩
var minifyCSS = require('gulp-minify-css');
var prefix = require('gulp-autoprefixer');
var sourcemaps = require('gulp-sourcemaps');

var livereload = require('gulp-livereload');  //livereload插件

// 语法检查  Lint JS
gulp.task('lint', function () {
    return gulp.src(['src/js/easeljs/*.js', '!src/js/main.js'])
        .pipe(jshint())
        .pipe(jshint.reporter('default'));
});


// 合并文件之后压缩代码
gulp.task('minify', function () {
    return gulp.src(['src/js/easeljs/easeljs-0.7.1.min.js', 'src/js/easeljs/tweenjs-0.5.1.min.js', 'src/js/easeljs/preloadjs-0.4.1.min.js', 'src/js/easeljs/movieclip-0.7.1.min.js', 'src/js/easeljs/soundjs-0.5.2.min.js', 'src/js/common.js', 'src/js/GameMain.js', 'src/js/hero.js', 'src/js/hero_assets.js'])
        .pipe(concat('main.js'))
        //.pipe(gulp.dest('src/js'))
        //.pipe(rename('main.js'))
        //.pipe(uglify({compress: {drop_console: true}}))
        .pipe(uglify())
        .pipe(gulp.dest('dist/js'));
});


//压缩HTML
//empty:不删除空的属性  cdata:脚本不带CDATA  comments:不删除注释  conditionals:不删除IE条件注释  spare:不删除冗余属性  quotes:不删除引号
gulp.task('minify-html', function () {
    var opts = {comments: false, spare: true, quotes: true, empty: false, conditionals: true};
    gulp.src('src/index.html')
        .pipe(minifyHTML(opts))
        .pipe(gulp.dest('dist/'))
});


//压缩CSS
gulp.task('minify-css', function () {
    gulp.src('src/css/*.css')
        //.pipe(prefix('last 2 versions', '> 1%', 'ie 9'))
        .pipe(minifyCSS({keepBreaks: true}))
        //.pipe(rename('style.min.css'))
        .pipe(gulp.dest('dist/css'))
});


//删除文件
gulp.task('clean', function (cb) {
    del(['dist/css', 'dist/js', 'dist/images'], cb)
});


// 监视文件的变化
gulp.task('watch', function () {
    gulp.watch('php/css_src/**/*.css', ['concat-css']);
    //gulp.watch('src/scss/**/*.scss', ['sass']);
    //gulp.watch('src/images/**/*', ['images']);

    //livereload.listen();
    //gulp.watch('dist/**').on('change', livereload.changed);
});


// 注册缺省任务
gulp.task('css', ['minify-css']);
gulp.task('default', ['minify', 'minify-css']);
gulp.task('debug', ['lint', 'minify', 'watch']);
